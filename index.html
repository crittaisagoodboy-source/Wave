window.addEventListener("message", (event) => { const msg = event.data; console.log("Received:", msg); if (msg === "scrollDown") { currentIndex = Math.min(buttons.length - 1, currentIndex + 1); updateHighlight(currentIndex); buttons[currentIndex]?.scrollIntoView({ block: "center" }); } else if (msg === "scrollUp") { currentIndex = Math.max(0, currentIndex - 1); updateHighlight(currentIndex); buttons[currentIndex]?.scrollIntoView({ block: "center" }); } else if (msg === "enter") { buttons[currentIndex]?.click(); } else if (msg === "tabLeft") { currentTabIndex = (currentTabIndex - 1 + tabButtons.length) % tabButtons.length; updateTabHighlight(currentTabIndex); } else if (msg === "tabRight") { currentTabIndex = (currentTabIndex + 1) % tabButtons.length; updateTabHighlight(currentTabIndex); } });
